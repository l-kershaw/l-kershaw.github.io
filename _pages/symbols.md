∈ : element of

⊆ : subseteq

⊇ : supseteq

⊊ : subsetneq

⊋ : supsetneq

× : times/multiplication

∘ : composition

→ : right arrow

← : left arrow

⇔ : if and only if

⇒ : implies

≠ : not equal to

≤ : less than or equal to

≥ : greater than or equal to

∪ : union

∩ : intersection

∅ : empty set

∧ : logical conjuction, "and"

∨ : logical disjunction, "or"
